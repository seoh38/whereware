-- EMPLOYEE 테이블 생성
CREATE TABLE EMPLOYEE (
    EMP_NO NUMBER PRIMARY KEY,
    JOB_CODE CHAR(2)DEFAULT 'J1',
    DEPT_CODE CHAR(2) DEFAULT 'D1',
    EMP_ID VARCHAR2(30) NOT NULL UNIQUE,
    EMP_PWD VARCHAR2(100) NOT NULL,
    EMP_NAME VARCHAR2(25) NOT NULL,
    EMP_EMAIL VARCHAR2(25),
    EMP_PHONE VARCHAR2(25),
    EMP_BIRTH VARCHAR2(25),
    EMP_ADDRESS VARCHAR2(100),
    SALARY NUMBER,
    HIRE_DATE DATE,
    END_DATE DATE,
    END_YN CHAR(1) DEFAULT 'N' CHECK (END_YN IN('Y', 'N')),
    ADMIN_YN CHAR(1) DEFAULT 'N' CHECK (ADMIN_YN IN('Y', 'N')),
    ENROLL_YN CHAR(1) DEFAULT 'N' CHECK (ENROLL_YN IN('Y', 'N')),
    ORIGINAL_PROFILENAME VARCHAR2(100),
    RENAMED_PROFILENAME VARCHAR2(100),
    STATUS CHAR(1) DEFAULT 'Y' CHECK (STATUS IN('Y', 'N'))
);

-- BOARD 테이블 생성
CREATE TABLE BOARD (
	BOARD_NO NUMBER,
	EMP_NO NUMBER,
    BOARD_WRITER VARCHAR2(20),
	BOARD_TITLE VARCHAR2(100),
	BOARD_CONTENT VARCHAR2(4000),
	BOARD_CREATED DATE DEFAULT SYSDATE,
	BOARD_MODIFIED DATE DEFAULT SYSDATE,
	BOARD_HITS NUMBER DEFAULT 0,
    COMMENT_COUNT NUMBER DEFAULT 0
);

-- BOARD_COMMENT 테이블 생성
CREATE TABLE BOARD_COMMENT (
	COMMENT_NO NUMBER,
	BOARD_NO NUMBER,
	EMP_NO NUMBER,
    COMMENT_WRITER VARCHAR2(20),
	COMMENT_CONTENT VARCHAR2(1000),
	B_COMMENT_CREATED DATE DEFAULT SYSDATE,
	B_COMMENT_MODIFIED DATE DEFAULT SYSDATE
);

-- BOARD_ATTACH 테이블 생성
CREATE TABLE BOARD_ATTACH(
    FILE_NO NUMBER PRIMARY KEY,
    BOARD_NO NUMBER,
    EMP_NO NUMBER,
    ORIGINAL_FILENAME VARCHAR2(100), 
	RENAMED_FILENAME VARCHAR2(100),
    FILE_UPLOADPATH VARCHAR2(200),
    FILE_SIZE NUMBER,
    FILE_TYPE CHAR(1) DEFAULT 'I',
    REG_DATE DATE
);



-- CMT 테이블
CREATE TABLE CMT_AT (
    CMT_NO NUMBER PRIMARY KEY,
    EMP_NO NUMBER NOT NULL,
    CMT_SRT DATE,
    CMT_END DATE,
    CMT_RS VARCHAR2(20),
    CMT_MONTH NUMBER,
    CMT_TIME VARCHAR2(10)
    );

-- CMT 휴식 테이블
CREATE TABLE CMT_RS (
    RS_NO NUMBER,
    RS_SRT DATE,
    RS_END DATE,
    CMT_NO NUMBER,
    EMP_NO NUMBER NOT NULL UNIQUE
    );


-- DOCUMENT 테이블
CREATE TABLE DOCUMENT (
	DOC_ID			NUMBER,
	EMP_NO			NUMBER NOT NULL,
	DOC_TITLE		VARCHAR2(500),
	DOC_CONTENT		VARCHAR2(2000),
	DOC_STATUS		VARCHAR2(200)	DEFAULT 'Y' CHECK (DOC_STATUS IN('Y', 'N')),
	DOC_EMERGENCY		VARCHAR2(100),
	DOC_DATE		DATE	DEFAULT SYSDATE,
	DOC_COMPLET_DATE	DATE
);

-- DOC_ATTACH 테이블
CREATE TABLE DOC_ATTACH (
	ATTACH_NUM	NUMBER,
	DOC_ID		NUMBER,
	ATTACH_ORIGIN	VARCHAR2(1000),
	ATTACH_RENAME	VARCHAR2(1000),
	ATTACH_EXT	VARCHAR2(100),
	ATTACH_SIZE	VARCHAR2(100)
);

-- REPORT_LINK 테이블
CREATE TABLE REPORT_LINK (
	LINK_NUM	NUMBER,
	DOC_ID		NUMBER NOT NULL,
	EMP_NO		NUMBER NOT NULL,
	LINK_NO		NUMBER,
	LINK_ID		VARCHAR2(10),
	LINK_DATE	DATE	DEFAULT SYSDATE,
	LINK_TYPE	CHAR(2) DEFAULT 'S1'
);

-- DOC_STATUS 테이블
CREATE TABLE DOC_STATUS (
    DOC_STATUS_CODE CHAR(2) PRIMARY KEY ,
    DOC_STATUS VARCHAR2(35)
);

-- 휴가 정보 테이블
CREATE TABLE DAYOFF (
    OFF_NO NUMBER PRIMARY KEY,
    EMP_NO NUMBER NOT NULL UNIQUE,
    OFF_SRT_DATE DATE NOT NULL,
    OFF_END_DATE DATE NOT NULL,
    OFF_REASON VARCHAR2(100) NOT NULL,
    OFF_YN CHAR(1) DEFAULT 'N' CHECK (OFF_YN IN('Y', 'N'))
);

-- 직급 테이블
CREATE TABLE JOB (
    JOB_CODE CHAR(2) PRIMARY KEY,
    JOB_NAME VARCHAR2(35)
);

-- 부서 테이블
--DROP TABLE DEPARTMENT;
CREATE TABLE DEPARTMENT (
    DEPT_ID CHAR(2) PRIMARY KEY,
    DEPT_TITLE VARCHAR2(35)
);

-- 캘린더 테이블
CREATE TABLE CALENDAR (
    	EMP_NO NUMBER,
	CALENDAR_NO NUMBER NOT NULL UNIQUE,
	CALENDAR_TITLE	VARCHAR2(200)	NOT NULL,
	CALENDAR_MEMO	VARCHAR2(500),
	CALENDAR_START	VARCHAR2(200),
	CALENDAR_END	VARCHAR2(200)
);
